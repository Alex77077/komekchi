import { GoogleGenerativeAI } from "@google/generative-ai";

export default async function handler(req, res) {
  // 1. Diňe POST kabul edýäris
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method Not Allowed" });
  }

  try {
    const apiKey = process.env.GEMINI_API_KEY;
    
    // 2. API Key barlagy
    if (!apiKey) {
      return res.status(500).json({ error: "API açary tapylmady! Vercel sazlamalaryny barlaň." });
    }

    const genAI = new GoogleGenerativeAI(apiKey);
    
    // 3. Model adyny iň ygtybarly usulda ýazýarys
    // Eger "models/gemini-pro" işlemese, diňe "gemini-pro" ýa-da "gemini-1.5-flash" synlaň
    const model = genAI.getGenerativeModel({ model: "gemini-pro" }); 

    const { message, systemPrompt } = req.body;

    // 4. AI-dan jogap alýarys
    const result = await model.generateContent(`${systemPrompt || ""}\n\nUlanyjy: ${message}`);
    const response = await result.response;
    const text = response.text();

    // 5. Jogaby ugradýarys
    return res.status(200).json({ reply: text });

  } catch (error) {
    console.error("Gemini API Error:", error);
    
    // Ýalňyşlyk bolsa-da JSON formatynda ugradýarys, şonda "JSON däl" diýen hata çykmaz
    return res.status(500).json({ 
      error: "AI-da näsazlyk döredi", 
      message: error.message 
    });
  }
}
